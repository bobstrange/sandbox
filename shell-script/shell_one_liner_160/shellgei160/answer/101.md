### 別解
```
別解1（田代）$ cat alphabet_connection | tr ' ' '\n' | ( cat; echo {a..z} | tr ' ' '\n' )| sort | uniq -c | awk '$1==1{print ""}$1==2{printf $2" "}END{print ""}' | awk 'NF==1{print}NF>1{print $1"-"$NF}' | xargs
別解2（eban）$ printf "%s\n" {a..z} $(<alphabet_connection) | sort | uniq -c | sed 's/.*2 //;/1/c\\n' | awk '{sub(/\n(.\n)*/,"-")}1' RS= | xargs
別解3（山田）$ cat alphabet_connection | perl -anle 'print unpack("C ", $_) for sort @F' | awk 'BEGIN{n=$0}{if($0 - n <= 1){printf $0" " }else{printf "\n"$0" "} n=$0}END{print}' | perl -anle 'printf pack("C", $_)."\t" for @F;print ""' | awk 'NF>=2{print $1"-"$NF}NF==1{print}' | xargs
別解4（上田）$ cat alphabet_connection | xargs -n 1 | sort | opy -v a=@ '[ "-"+F1 if ord(a)+1 == ord( F1) else F1 ];{a=F1}' | xargs | sed -E 's/([^-])( -.)* -(.)/\1-\3/g'
```
